BITS : 16

MAXCALL : 32

MAXDATA : 128

ADD (A,B,RETURN) : {
RGA A
RGB B
ADD RETURN
}

NOR (A,B,RETURN) : {
RGA A
RGB B
NOR RETURN
}

LOAD (RAM_ADDRESS,DESTINATION) : {
MAR RAM_ADDRESS
LOD DESTINATION
}

STORE (SOURCE,RAM_ADDRESS) : {
MAR RAM_ADDRESS
STR SOURCE
}

JUMP IF GREATER (A,B,ADDRESS) : {
RGA A
RGB B
JMP GRAT ADDRESS
}

IMM (VALUE,DESTINATION) : {
IMM VALUE DESTINATION
}

RIGHT SHIFT (SOURCE,RETURN) : {
RGA SOURCE
RSH RETURN
}
